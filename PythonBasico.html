"""**2 Mesma letra**:

Escreva uma função que receba uma string com duas palavras e

retorne True se ambas palavras começarem com a mesma letra.

Exemplo:

* mesma_letra('Cão covarde') -> True

* mesma_letra('Vira Lata') -> False

"""

#Exercicio02

def mesma_letra(text) :

  words = text.lower().split(' ')

  prevStartLetter = ''

  for word in words :

    curStartLetter = word[0]

   

    if(prevStartLetter != '' and prevStartLetter != curStartLetter) :

      return False

     

    prevStartLetter = curStartLetter

  return True

#main program

print(mesma_letra('Cão covarde'))

print(mesma_letra('Vira Lata'))

"""**3 Mestre Yoda**:

Dada uma sentença, a função deve retornar a sentença com as

palavras na ordem reversa.

Exemplo:

* mestre_yoda('Eu estou em casa') --> 'casa em estou Eu'

* mestre_yoda('Estamos prontos') --> 'prontos Estamos'

"""

#Exercicio03

def mestre_yoda(text) :

  words = text.split(' ')

  finalPhrase = ''

  for i in range(len(words) -1,-1, -1) :

    finalPhrase += words[i] + ' '

   

  return finalPhrase.strip()

#main program

print(mestre_yoda('Eu estou em casa'))

print(mestre_yoda('Estamos prontos'))

  sum = 0

  for number in list :

    sum += number

   

  return sum

  for i in range(0, len(list)) :

    if(list[i] == number) : return i

   

  return -1

def blackjack(num1, num2, num3) :

  list = [num1, num2, num3]

  sumV = sum(list)

 

  if(sumV < 21) : return sumV

  if(sumV > 21 and findNumber(11, list) != -1) :

    sumV -= 10

 

  return 'Estourou' if sumV > 21 else sumV

"""**6 Espião**:

Escreva uma função que receba uma lista de

inteiros e retorne True *se contém um 007 em ordem*, mesmo

que não contínuo.

Exemplo:

* espiao([1,2,4,0,0,7,5]) --> True

* espiao([1,0,2,4,0,5,7]) --> True

* espiao([1,7,2,4,0,5,0]) --> False

"""

#Exercicio06

def espiao(list) :

  expected = [0,0,7]

  expectedIndex = 0

 

  for number in list :

    if (number == expected[expectedIndex]) :

      expectedIndex += 1

     

    if(expectedIndex == len(expected)) : return True

 

  return False

#main program

print(espiao([1,2,4,0,0,7,5]))

print(espiao([1,0,2,4,0,5,7]))
